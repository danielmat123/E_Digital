import serial   # pip install pyserial
import time


# =============================================================
# CONFIGURACIÓN — ajusta PUERTO_SERIAL si es necesario
# =============================================================
PUERTO_SERIAL = "COM8"   # <-- verificar y corregir si es diferente
BAUDRATE      = 9600
TIMEOUT_S     = 2


# =============================================================
# FUNCIONES AUXILIARES
# =============================================================

def enviar_y_recibir(puerto, trama):
    """
    Envía una trama al Arduino y devuelve la respuesta cruda.
    Parámetros:
        puerto — objeto serial.Serial abierto
        trama  — string en formato compacto, ej: "ST 00000"
    Devuelve:
        string con la respuesta cruda del Arduino, ej: "OK 10100"
    """
    puerto.write((trama + "\n").encode("utf-8"))
    respuesta = puerto.readline().decode("utf-8", errors="replace").strip()
    return respuesta


def mostrar_menu():
    """Muestra las opciones disponibles al usuario."""
    print("\n--- MENÚ ---")
    print("  1. Ver estado del sistema")
    print("  2. Encender LED")
    print("  3. Apagar LED")
    print("  4. Parpadear LED")
    print("  5. Ver eventos del botón")
    print("  0. Salir")
    print("------------")

# PROGRAMA PRINCIPAL
# =============================================================

    try:

        print(f"Detalle: {e}")
